   in P_NO_NOTA VARCHAR(8)

CREATE PROCEDURE SP_UPDATE_NOTAS(
  IN P_NO_NOTA VARCHAR(8),
  IN P_TIPO_NOTA VARCHAR(10)
)
LANGUAGE SQL
NO EXTERNAL ACTION
DETERMINISTIC
BEGIN
    UPDATE NOTAS SET TIPO_NOTA = P_TIPO_NOTA WHERE NO_NOTA = P_NO_NOTA;
END;
